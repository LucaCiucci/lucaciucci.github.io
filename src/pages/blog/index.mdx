import { getCollection } from 'astro:content';
import { Code } from 'astro:components';
import { Md } from '@luca/astro-ui/components';

export const all_posts = await getCollection('blog_posts');

export function post_entry(post) {
  const main = <>{post.data.date ? `${post.data.date.toISOString().split('T')[0] }: ` : ''}<a href={`/blog/${post.id}`}><Md kind="inline" text={post.data.title || 'Untitled'} /></a>{post.data.subtitle && <span style={{
    color: 'var(--text-muted)',
  }}> — <Md kind="inline" text={post.data.subtitle} /></span>}</>;
  return (<li>
    {post.data.description ? <details>
      <summary>{main}</summary>
      <Md text={post.data.description} />
    </details> : main}
  </li>);
}
