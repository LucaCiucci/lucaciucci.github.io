import { PageIntro, ProjectCard, ProjectGrid } from '@luca/astro-ui/components';
import all_projects_data from './_all.yaml';
import { mergeProjects } from '../../utils/projects';

export const all_projects = await mergeProjects(all_projects_data.projects);

<PageIntro
  eyebrow="Project archive"
  title="Projects"
  description="Experiments, software, research work, and things built to answer questions that would not leave me alone."
/>

export function render_card(project) {
  return (<ProjectCard
    title={project.title}
    description={project.description}
    href={project.href}
    meta={project.date}
    tags={project.tags}
    status={project.status}
    image={project.image}
    imageAlt={project.imageAlt}
  />)
}

<ProjectGrid>
  {all_projects.filter(project => project.featured).sort((a, b) => a.featured - b.featured).map(render_card)}
</ProjectGrid>
